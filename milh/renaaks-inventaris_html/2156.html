98             _tabel = new DefaultTableModel(null,kolom){
99                 Class[] types = new Class [] {
100                    java.lang.String.class,
101                    java.lang.String.class,
102                    java.lang.String.class,
103                    java.lang.String.class,
104                    java.lang.String.class
105                 };
106                 public Class getColumnClass(int columnIndex) {
107                     return types [columnIndex];
108                 }
109                 // Agar table tidak bisa diedit
110                 public boolean isCellEditable(int row, int col) {
111                     int cola = _tabel.getColumnCount();
112                     return (col < cola) ? false : true;
113                 }
114             };
115             view.getTabelpermohonanpemohon().setModel(_tabel);
116             try{
117                 HapusTabel();
118                 String sql = "" + "SELECT * from permohonan where id_Pemohon = '"+id+"'";
119                 ResultSet rs = statement.executeQuery(sql);
120                 while(rs.next()){
121                     String Kode_Permohonan = rs.getString(1);
122                     String Kode_Barang = rs.getString(2);
123                     int Jumlah_Barang = rs.getInt(4);

125                     String Status = rs.getString(6);
126                     Object[] data = {Kode_Permohonan,Kode_Barang,Jumlah_Barang,Keterangan,Status};

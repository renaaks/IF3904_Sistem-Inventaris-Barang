6     package controller;
7     
8     import model.Aplikasi;
9     import model.Connect;
10     import view.LaporanLogistik;
11     import java.awt.event.ActionEvent;
12     import java.awt.event.ActionListener;
13     
14     /**
15      *
16      * @author Novelty Octaviani
17      */
18     public class ControlerLaporanLogistik implements ActionListener{
19         private Aplikasi model;
20         private LaporanLogistik view;
21         private Connect data=new Connect();
22     
23         
24         ControlerLaporanLogistik(Aplikasi model){
25             this.model = model;
26             data.tryConnect();
27             view = new LaporanLogistik();
28             view.setVisible(true);
29             view.addListener((ActionListener) this);
30             model.showTabelBarangMasukLaporanLogistik(view);
31             model.showTabelBarangKeluarLaporanLogistik(view);
32         }
33         
34          
35           
36         
37         @Override
38         public void actionPerformed(ActionEvent ae) {
39            Object source = ae.getSource();
40            if (source == view.getBtnbacklaporanlogistik()){
41                new ControlerMenuLogistik(model);
42                view.dispose();
43            } else if (source == view.getBtnlogoutlaporanlogistik()){
44                new ControlerLogin(model);
45                view.dispose();
46            }
47         }
48     }

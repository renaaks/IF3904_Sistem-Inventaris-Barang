6     package Controller;
7     
8     import Database.Connect;
9     import Model.Aplikasi;
10     import Model.Barang;
11     import View.DataBarangLogistik;
12     import java.awt.event.ActionEvent;
13     import java.awt.event.ActionListener;
14     import java.util.Random;
15     import javax.swing.JOptionPane;
16     
17     /**
18      *
19      * @author Novelty Octaviani
20      */
21     public class ControlerDataBarangLogistik implements ActionListener{
22         private Aplikasi model;
23         private DataBarangLogistik view;
24         private Connect data=new Connect();
25         
26         ControlerDataBarangLogistik(Aplikasi model){
27             this.model = model;
28             data.tryConnect();
29             view = new DataBarangLogistik();
30             view.setVisible(true);
31             view.addListener(this);
32             Random r = new Random();
33             int randId = r.nextInt(999-111) + 111;
34             view.getTfkodebarangdatabaranglogistik().setText("BRG"+Integer.toString(randId));
35             model.showTabelBarang(view);
36         }
37     
38         @Override
39         public void actionPerformed(ActionEvent ae) {
40             Object source = ae.getSource();
41             if (source == view.getBtnadddatabaranglogistik()){
42                 if (view.getTfkodebarangdatabaranglogistik().getText().equals("")||view.getTfjenisbarangdatabaranglogistik().getText().equals("")||view.getTfjumlahbarangdatabaranglogistik().getText().equals("")||view.getTfnamabarangdatabaranglogistik().getText().equals("")){
43                     JOptionPane.showMessageDialog(null, "lengkapi data terlebih dahulu");
44                 }
45                 else{
46                     String kodebarang = view.getTfkodebarangdatabaranglogistik().getText();
47                     String namabarang = view.getTfnamabarangdatabaranglogistik().getText();
48                     String jenisbarang = view.getTfjenisbarangdatabaranglogistik().getText();
49                     int jumlahbarang = Integer.parseInt(view.getTfjumlahbarangdatabaranglogistik().getText());
50                     Barang brg = new Barang(kodebarang, namabarang, jenisbarang, jumlahbarang);
51                     model.addbarang(brg);
52                     model.showTabelBarang(view);
53                 }
54             view.refresh("");
55             } else if (source == view.getBtnbackdatabaranglogistik()){
56                 new ControlerMenuLogistik(model);
57                 view.dispose();
58             } else if (source == view.getBtnlogoutdatabaranglogistik()){
59                 new ControlerLogin(model);
60                 view.dispose();
61             }
62         view.refresh("");
63         }
64         
65     }

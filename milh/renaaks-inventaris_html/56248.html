6     package Controller;
7     
8     import Database.Connect;
9     import Model.Aplikasi;
10     import Model.Barang_Keluar;
11     import Model.Barang_Masuk;
12     import Model.Pemohon;
13     import View.PendataanBarang;
14     import java.awt.event.ActionEvent;
15     import java.awt.event.ActionListener;
16     import java.util.Random;
17     import javax.swing.JOptionPane;
18     
19     /**
20      *
21      * @author Novelty Octaviani
22      */
23     public class ControlerPendataanBarang implements ActionListener{
24         
25         private Aplikasi model;
26         private PendataanBarang view;
27         private Connect data=new Connect();
28         private Pemohon p;
29         
30         ControlerPendataanBarang (Aplikasi model, Pemohon p){
31             this.p = p;
32             this.model = model;
33             data.tryConnect();
34             view = new PendataanBarang();
35             view.setVisible(true);
36             view.addListener(this);
37             Random r = new Random();
38             int randId = r.nextInt(999-111) + 111;
39             view.getTfbarangkeluarpemohon().setText("BRGKLR"+Integer.toString(randId));
40             model.showTabelBarangKeluarPemohon(view,p.getId_Pemohon());
41             model.showTabelBarangMasukPemohon(view,p.getId_Pemohon());
42         }
43           
44         @Override
45         public void actionPerformed(ActionEvent ae) {
46             Object source = ae.getSource();
47             if (source == view.getBtnaddbarangkeluarpemohon()){
48             if (view.getTfkodebarangmasukpemohon().getText().equals("")||view.getTfkondisibarangpemohon().getText().equals("")||view.getTftanggalkeluarpemohon().getText().equals("")||view.getTfjumlahbarangkeluar().getText().equals("")){
49                    JOptionPane.showMessageDialog(null, "lengkapi data terlebih dahulu");
50                }
51                else {
52                    String kode_barang_keluar = view.getTfbarangkeluarpemohon().getText();
53                    String kode_barang_masuk = view.getTfkodebarangmasukpemohon().getText();
54                    int jumlah_barang = Integer.parseInt(view.getTfjumlahbarangkeluar().getText());
55                    String kondisi_barang = view.getTfkondisibarangpemohon().getText();
56                    String tanggal_keluar = view.getTftanggalkeluarpemohon().getText();
57                    if (model.carikodebarangmasuk(kode_barang_masuk) != null){ //cari kode barang
58                        //cek apakah jumlah barang mencukupi
59                         Barang_Masuk brg = model.carikodebarangmasuk(kode_barang_masuk);
60                         if (brg.getJumlah_barang()>=jumlah_barang) {
61                             int sisaBarang = brg.getJumlah_barang()-jumlah_barang;
62                             model.updateBarangMasuk(kode_barang_masuk, sisaBarang);
63                             Barang_Keluar brgkeluar = new Barang_Keluar (kode_barang_masuk, kode_barang_keluar, p.getId_Pemohon(), jumlah_barang, kondisi_barang, tanggal_keluar);
64                             model.addbarangkeluar(brgkeluar);
65                             model.showTabelBarangKeluarPemohon(view,p.getId_Pemohon());
66                             model.showTabelBarangMasukPemohon(view, p.getId_Pemohon());
67                         } else {
68                             JOptionPane.showMessageDialog(null, "Jumlah data barang yang tersedia tidak mencukupi");
69                         }
70                    }else {
71                    JOptionPane.showMessageDialog(null, "Kode barang yang anda masukan tidak sesuai");
72                    }
73                }
74                
75             } else if (source == view.getBtnbackpendataanbarangpemohon()){
76                 new ControlerMenuPemohon(model,p);
77                 view.dispose();
78             } else if (source == view.getBtnlogoutpendataanbarangpemohon()){
79                 new ControlerLogin(model);
80                 view.dispose();
81             }
82             view.refresh("");
83         }
84     }
